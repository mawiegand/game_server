def self.quad_tree_to_tms_tile_code(path)
  zoom = path.length
  tx = 0
  ty = 0
              
              zoom.downto(1) do |i|
                digit = Integer(path[zoom-i])
                      mask = 1 << (i-1)
                      if (digit & 1) != 0 
                              tx += mask
                      end
                      if (digit & 2) != 0 
                              ty += mask
                      end
  end
              
              ty = (1 << zoom) - 1 - ty
              return { x: tx, y: ty, zoom: zoom }
end

def self.tms_to_quad_tree_tile_code(tx, ty, zoom)
  path = ""
  ty = (1<<zoom)-ty-1
  
  zoom.downto(1) do |i|
    digit = 0
    mask = 1 << (i-1)
    if (tx & mask) != 0
      digit+= 1;
    end
    if (ty & mask) != 0 
      digit+= 2;
    end
    path = "#{path}#{digit}"
  end
  
  return path
end
